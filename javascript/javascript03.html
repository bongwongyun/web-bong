16. 콜백 함수

{
    const info = {
        Num1: 1,
        Name1: "why",
        Job1: "if",
        Num2: 2,
        Name2: "왜",
        Job2: "만약에",

        result1: function () {
            document.write(this.Num1 + ". 내 이름은" + this.Name1 + "이며 직업은" + this.Job1 + "입니다.");
        },
        result2: function () {
            document.write(this.Num2 + ". 내 이름은" + this.Name2 + "이며 직업은" + this.Job2 + "입니다.");
        }
    }
    info.result1();
    info.result2();   
}

결과보기

1. 내 이름은why이며 직업은if입니다.2. 내 이름은왜이며 직업은만약에입니다.

17. 객체 생성자 함수

{
    function func(num, name, job) {
        this.num = num;
        this.name = name;
        this.job = job;
        
        this.result = function () {
            document.write(this.num + ". 내 이름은" + this.name + "이며 직업은" + this.job + "입니다.");
        };
    };
        
    //인스턴스 생성
    const info1 = new func("1", "나는요", "학자에요");
    const info2 = new func("2", "학과는", "@@");
        
    info1.result();
    info2.result(); 
}

결과보기

1. 내 이름은나는요이며 직업은학자에요입니다.2. 내 이름은학과는이며 직업은@@입니다.

18. 프로토타입 함수

{
    function func(num, name, job) {
        this.num = num;
        this.name = name;
        this.job = job;
    };
    func.prototype.result = function () {
        document.write(this.num + ". 내 이름은" + this.name + "이며 직업은" + this.job + "입니다.");

    //인스턴스 생성
    const info1 = new func("1", "나는요", "학자에요");
    const info2 = new func("2", "학과는", "@@@");
        
    info1.result();
    info2.result();
}

결과보기

1. 내 이름은나는요이며 직업은학자에요입니다.2. 내 이름은학과는이며 직업은@@@입니다.

19. 객체 리터럴 함수

{
    function func(num, name, job) {
        this.num = num;
        this.name = name;
        this.job = job;
    }
    func.prototype = {
        result1: function () {
            document.write(this.num + ". 내 이름은 " + this.name + "이며, 직업은 " + this.job + "입니다. 
");
        },
        result2: function () {
            document.write(this.num + ". 내 이름은 " + this.name + "이며, 직업은 " + this.job + "입니다. 
");
        }
    }
    //인스턴스 생성
    const info1 = new func("1", "날강두", "감독");
    const info2 = new func("2", "꾸짖을 갈", "호통");
        
    info1.result1();
    info2.result2();
}

결과보기

1. 내 이름은 날강두이며, 직업은 감독입니다.2. 내 이름은 꾸짖을 갈이며, 직업은 호통입니다.

20. 클래스 : 함수의 집합체

{
    class study {
        constructor(num, name, job) {
            this.num = num;
            this.name = name;
            this.job = job;
        }
        result() {
            document.write(this.num + ". 내 이름은" + this.name + "이며 직업은" + this.job + "입니다.");
        }
    }
    //인스턴스 생성
    const info1 = new study("1", "김우진", "학자");
    const info2 = new study("2", "오대수", "교수");
        
    info1.result();
    info2.result();
}

결과보기

1. 내 이름은김우진이며 직업은학자입니다.2. 내 이름은오대수이며 직업은교수입니다.
