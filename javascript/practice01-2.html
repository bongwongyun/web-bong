2. 2단부터 9단까지 출력하기


2.1 for 문을 이용해서 2단부터 9단까지 출력해주세요.

{             
    for (let i=2; i<=9; i++){
        document.write("<br>");
           
        for (let j=1; j<=9; j++){
            let sum = i * j; 
               document.write( i + " * " + j + " = " + sum, "<br>");
        }
    }
}

{
    let i = 1;
    while(i<=8){
      document.write("<br>");
        i++
          let j=1;
          while(j<=9){
